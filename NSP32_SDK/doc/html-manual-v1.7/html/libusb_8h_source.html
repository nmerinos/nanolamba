163 static inline uint16_t libusb_cpu_to_le16(const uint16_t x)
164 {
165  union {
166  uint8_t b8[2];
167  uint16_t b16;
168  } _tmp;
169  _tmp.b8[1] = (uint8_t) (x >> 8);
170  _tmp.b8[0] = (uint8_t) (x & 0xff);
171  return _tmp.b16;
172 }
